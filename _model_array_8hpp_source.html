288 // Generic index generator that will work on any container
289 template <typename T, typename C> static T indexrHelper(const T* dims, const C& loc)
290 {
291 size_t ndims = loc.size();
292 T stride = 1;
293 T ii = 0;
294 auto iloc = rbegin(loc);
295 for (size_t dim = ndims; dim > 0; --dim) {
296 ii += stride * (*iloc++);
297 stride *= dims[dim - 1];
298 }
299 return ii;
300 }
